let isDrawing=false,penSize=2.5,eraserMode=false,lastPos=null,lastMidPos=null;

// penOpts wired in DOMContentLoaded below

function getPos(e){
const r=inkCanvas.getBoundingClientRect();
const src=e.touches?e.touches[0]:e;
return{x:(src.clientX-r.left)*(CUR_W/r.width),y:(src.clientY-r.top)*(CUR_H/r.height)};
}
function startDraw(e){
e.preventDefault();
isDrawing=true;
lastPos=getPos(e);
lastMidPos=null;
// Draw a dot for single taps
if(!eraserMode){
inkCtx.globalCompositeOperation='source-over';
inkCtx.fillStyle='rgba(8,5,2,.97)';
inkCtx.beginPath();
inkCtx.arc(lastPos.x,lastPos.y,penSize/2,0,Math.PI*2);
inkCtx.fill();
}
}
function moveDraw(e){
e.preventDefault();
if(!isDrawing)return;
const pos=getPos(e);

if(eraserMode){
inkCtx.globalCompositeOperation='destination-out';
inkCtx.strokeStyle='rgba(255,255,255,1)';
inkCtx.lineWidth=penSize*5;
inkCtx.lineCap='round';inkCtx.lineJoin='round';
inkCtx.beginPath();inkCtx.moveTo(lastPos.x,lastPos.y);inkCtx.lineTo(pos.x,pos.y);inkCtx.stroke();
inkCtx.globalCompositeOperation='source-over';
}else{
// Smooth quadratic bezier for natural strokes
inkCtx.globalCompositeOperation='source-over';
inkCtx.strokeStyle='rgba(8,5,2,.97)';
inkCtx.lineWidth=penSize;
inkCtx.lineCap='round';inkCtx.lineJoin='round';

const midX=(lastPos.x+pos.x)/2;
const midY=(lastPos.y+pos.y)/2;

inkCtx.beginPath();
if(lastMidPos){
  inkCtx.moveTo(lastMidPos.x,lastMidPos.y);
  inkCtx.quadraticCurveTo(lastPos.x,lastPos.y,midX,midY);
}else{
  inkCtx.moveTo(lastPos.x,lastPos.y);
  inkCtx.lineTo(midX,midY);
}
inkCtx.stroke();
lastMidPos={x:midX,y:midY};

}
lastPos=pos;
}
function endDraw(e){
e.preventDefault();
if(isDrawing&&lastPos&&lastMidPos&&!eraserMode){
// Finish the last segment
inkCtx.beginPath();
inkCtx.moveTo(lastMidPos.x,lastMidPos.y);
inkCtx.lineTo(lastPos.x,lastPos.y);
inkCtx.stroke();
}
isDrawing=false;lastPos=null;lastMidPos=null;
inkCtx.globalCompositeOperation='source-over';
}

function extractInk(){
const w=CUR_W,h=CUR_H;
const id=inkCtx.getImageData(0,0,w,h);const d=id.data;
let x0=w,x1=-1,y0=h,y1=-1;
for(let y=0;y<h;y++)for(let x=0;x<w;x++){
const i=(y*w+x)*4;
const r=d[i],g=d[i+1],b=d[i+2],a=d[i+3];
if(a<15||r>210&&g>210&&b>210)continue;
if(x<x0)x0=x;if(x>x1)x1=x;if(y<y0)y0=y;if(y>y1)y1=y;
}
if(x1<0||y1<0)return null;
const pad=4;
const cx=Math.max(0,x0-pad),cy=Math.max(0,y0-pad);
const cw=Math.min(w,x1+pad+1)-cx,ch=Math.min(h,y1+pad+1)-cy;
if(cw<=0||ch<=0)return null;
const tmp=document.createElement('canvas');tmp.width=cw;tmp.height=ch;
const tc=tmp.getContext('2d');
const od=tc.createImageData(cw,ch);const o=od.data;
for(let y=0;y<ch;y++)for(let x=0;x<cw;x++){
const si=((cy+y)*w+(cx+x))*4;const di=(y*cw+x)*4;
const r=d[si],g=d[si+1],b=d[si+2],a=d[si+3];
if(a>=15&&!(r>210&&g>210&&b>210)){o[di]=r;o[di+1]=g;o[di+2]=b;o[di+3]=a;}
}
tc.putImageData(od,0,0);
const ct=getCharType(selectedChar);
let glyphH,aboveBase,belowBase;
if(ct){
glyphH=ct.glyphH;aboveBase=ct.aboveBase;belowBase=ct.belowBase;
}else{
aboveBase=Math.max(0.1,(CUR_BASELINE-cy)/CUR_XH);
belowBase=Math.max(0,(cy+ch-CUR_BASELINE)/CUR_XH);
glyphH=aboveBase+belowBase;
}
return{dataURL:tmp.toDataURL('image/png'),glyphH,aboveBase,belowBase,aspect:cw/ch,v:8};
}

// ═══════════════════════════════════════════════════════════════
// STROKE THICKENING – only changes line weight, not glyph size
// This runs on extracted ink data at render time.
// ═══════════════════════════════════════════════════════════════

function getTintedImage(img,hexColor,strokeScale){
strokeScale=strokeScale||1.0;
var ssKey=Math.round(strokeScale*10);
var key=img.src+'|'+hexColor+'|'+ssKey;
if(tintCache.has(key))return tintCache.get(key);
const r=parseInt(hexColor.slice(1,3),16);
const g=parseInt(hexColor.slice(3,5),16);
const b=parseInt(hexColor.slice(5,7),16);
const tw=img.naturalWidth,th=img.naturalHeight;
const tmp=document.createElement('canvas');tmp.width=tw;tmp.height=th;
const tc=tmp.getContext('2d');
tc.drawImage(img,0,0);
const id=tc.getImageData(0,0,tw,th);const dd=id.data;
for(let i=0;i<dd.length;i+=4){
if(dd[i+3]>10){
const brightness=(dd[i]+dd[i+1]+dd[i+2])/3;
const t=brightness/255;
dd[i]=Math.round(r+(255-r)*t);
dd[i+1]=Math.round(g+(255-g)*t);
dd[i+2]=Math.round(b+(255-b)*t);
// Ensure full opacity for dark pixels – no weak spots
if(brightness<180){dd[i+3]=Math.min(255,Math.round(dd[i+3]*1.15));}
}
}
tc.putImageData(id,0,0);
// Apply stroke thickness BEFORE caching
if(strokeScale>1.08){
var rad=Math.max(1,Math.round((strokeScale-1.0)*2.0));
var dilated=document.createElement('canvas');dilated.width=tw;dilated.height=th;
var dc2=dilated.getContext('2d');
for(var dy=-rad;dy<=rad;dy++){for(var dx=-rad;dx<=rad;dx++){
if(dx*dx+dy*dy<=rad*rad+1)dc2.drawImage(tmp,dx,dy);
}}
var id2=dc2.getImageData(0,0,tw,th);var d2=id2.data;
for(var j=0;j<d2.length;j+=4){if(d2[j+3]>30)d2[j+3]=255;}
dc2.putImageData(id2,0,0);
tintCache.set(key,dilated);return dilated;
}
tintCache.set(key,tmp);return tmp;
}

async function generateText(){
const hw=getActiveHw();if(!hw){toast('Keine Handschrift aktiv!');return;}
const text=document.getElementById('inputText').value;if(!text.trim()){toast('Bitte Text eingeben!');return;}
const charSize=+document.getElementById('charSize').value;
const charSpacing=+document.getElementById('charSpacing').value;
const wordMult=+document.getElementById('wordSpacing').value;
const lineMult=+document.getElementById('lineSpacingMult').value;
const randomMode=document.getElementById('randomMode').value;
const paperStyle=document.getElementById('paperStyle').value;
const inkColor=document.getElementById('inkColor').value;
const strokeScaleEl=document.getElementById('strokeScale');
const strokeScale=strokeScaleEl?(+strokeScaleEl.value||1.0):1.0;
// Unsauberkeiten: randomly omit commas (only when >100 words, max 2.9/100)
const errorOn=(document.getElementById('errorToggle')&&document.getElementById('errorToggle').classList.contains('on'))||false;
const errorRate=errorOn?(+(document.getElementById('errorRate')?document.getElementById('errorRate').value:1.5)):0;
const jitterOn=document.getElementById('jitterToggle').classList.contains('on');
const jitterAmt=jitterOn?(+document.getElementById('jitterAmt').value):0;
const slantOn=document.getElementById('slantToggle').classList.contains('on');
const slantBaseDeg=slantOn?(+document.getElementById('slantDeg').value):0;
const slantVar=slantOn?(+document.getElementById('slantVar').value):0;
const lineH=Math.round(charSize*lineMult);
const wordSpace=Math.round(charSize*wordMult);

tintCache.clear();
const wordCountTotal=(text.match(/\S+/g)||[]).length;

const needed=new Set([...text].filter(c=>c!==' '&&c!=='\n'));
const missing=new Set();
const cache={};

function parseVariant(pick,nW,nH){
let dH,dW,blY;
if(pick.glyphH!=null){
dH=Math.max(1,Math.round(pick.glyphH*charSize));
dW=Math.max(1,Math.round(dH*(pick.aspect||nW/nH)));
blY=Math.round(pick.aboveBase*charSize);
}else if(pick.aboveBase!=null){
dH=Math.max(1,Math.round((pick.aboveBase+pick.belowBase)*charSize));
dW=Math.max(1,Math.round(dH*(pick.aspect||nW/nH)));
blY=Math.round(pick.aboveBase*charSize);
}else{
dH=Math.round(charSize*1.35);dW=Math.round(nW*(dH/nH));blY=Math.round(dH*0.75);
}
return{dW:Math.max(1,dW),dH:Math.max(1,dH),blY};
}

await Promise.all([...needed].map(c=>new Promise(resolve=>{
const variants=hw.chars[c]||[];
if(!variants.length){missing.add(c);resolve();return;}
const vlist=randomMode==='random'?variants:[variants[0]];
let loaded=0;const parsed=[];
vlist.forEach(pick=>{
const src=typeof pick==='string'?pick:pick.dataURL;
if(!src){loaded++;if(loaded===vlist.length){if(parsed.length)cache[c]=parsed;else missing.add(c);resolve();}return;}
const img=new Image();
img.onload=()=>{
const nW=img.naturalWidth,nH=img.naturalHeight;
if(nW&&nH){parsed.push({img,...parseVariant(pick,nW,nH)});}
loaded++;
if(loaded===vlist.length){if(parsed.length)cache[c]=parsed;else missing.add(c);resolve();}
};
img.onerror=()=>{loaded++;if(loaded===vlist.length){if(parsed.length)cache[c]=parsed;else missing.add(c);resolve();}};
img.src=src;
});
})));

function pickVariant(c){
const arr=cache[c];if(!arr||!arr.length)return null;
return arr[Math.floor(Math.random()*arr.length)];
}

// Word-wrap
const finalLines=[];
const lines=text.split('\n').map(l=>l.trimEnd());
for(const raw of lines){
if(!raw.trim()){finalLines.push([]);continue;}
let curLine=[],curW=0;
for(const word of raw.split(' ').filter(w=>w.length>0)){
const tokens=[...word].reduce((acc,ch)=>{
const arr=cache[ch];
if(!arr||!arr.length)return acc;
const ref=arr[0];
acc.push({type:'char',ch,dW:ref.dW});return acc;
},[]);
if(!tokens.length)continue;
const wordW=tokens.reduce((s,t,i)=>s+t.dW+(i>0?charSpacing:0),0);
if(curLine.length>0&&curW+wordSpace+wordW>A4_TW){
finalLines.push(curLine);curLine=[...tokens];curW=wordW;
}else{
if(curLine.length>0){curLine.push({type:'space',width:wordSpace});curW+=wordSpace;}
curLine.push(...tokens);
curW=0;let pc=false;
for(const t of curLine){if(t.type==='space'){curW+=t.width;pc=false;}else{if(pc)curW+=charSpacing;curW+=t.dW;pc=true;}}
}
}
if(curLine.length)finalLines.push(curLine);
}

const firstBaseline=A4_MT+lineH;
const linesPerPage=Math.max(1,Math.floor((A4_H-firstBaseline-A4_MB)/lineH));
const totalPages=Math.max(1,Math.ceil(finalLines.length/linesPerPage));

pageCanvases=[];
const wrap=document.getElementById('pagesWrap');wrap.innerHTML='';
const pngBtns=document.getElementById('pngBtns');pngBtns.innerHTML='';

const DPR=2;

for(let page=0;page<totalPages;page++){
const canvas=document.createElement('canvas');
canvas.width=RW;canvas.height=RH;
const ctx=canvas.getContext('2d');
ctx.scale(DPR,DPR);
drawA4Paper(ctx,paperStyle,lineH,firstBaseline);

const pageLines=finalLines.slice(page*linesPerPage,(page+1)*linesPerPage);
for(let li=0;li<pageLines.length;li++){
  const baseline=firstBaseline+li*lineH;
  let x=A4_ML,pc=false;
  for(const tok of pageLines[li]){
    if(tok.type==='space'){x+=tok.width;pc=false;continue;}
    if(pc)x+=charSpacing;
    const v=pickVariant(tok.ch);
    if(!v){x+=tok.dW;pc=true;continue;}
    if(errorRate>0&&tok.ch===','&&wordCountTotal>100){
      if(Math.random()<(errorRate/100)){pc=true;continue;}
    }
    
    // Jitter: subtle baseline shift
    let jitter=0;
    if(jitterAmt>0){
      const mag=Math.pow(Math.random(),1.4)*jitterAmt;
      jitter=Math.random()<0.70?-mag:mag*0.5;
    }
    const drawY=Math.round(baseline-v.blY+jitter);
    
    // Tint with stroke thickness baked in – glyph dimensions UNCHANGED
    const tinted=getTintedImage(v.img,inkColor,strokeScale);
    
    // Slant per character
    const slantThisChar=slantBaseDeg+(slantVar>0?(Math.random()-0.5)*2*slantVar:0);
    const slantRad=slantThisChar*Math.PI/180;
    
            if(slantRad!==0){
      ctx.save();
      ctx.transform(1,0,Math.tan(slantRad),1,-baseline*Math.tan(slantRad),0);
      ctx.drawImage(tinted,x,drawY,v.dW,v.dH);
      ctx.restore();
    }else{
      ctx.drawImage(tinted,x,drawY,v.dW,v.dH);
    }
    x+=v.dW;pc=true;
  }
}
pageCanvases.push(canvas);

const availW=Math.max(300,(document.getElementById('pagesWrap').clientWidth||700));
const dispScale=availW/A4_W;
const dpr=Math.min(window.devicePixelRatio||1,2);
const dispW=Math.round(A4_W*dispScale*dpr);
const dispH=Math.round(A4_H*dispScale*dpr);
const dispCanvas=document.createElement('canvas');
dispCanvas.width=dispW;dispCanvas.height=dispH;
dispCanvas.style.cssText='width:100%;height:auto;display:block;';
const dc=dispCanvas.getContext('2d');
dc.imageSmoothingEnabled=true;
dc.imageSmoothingQuality='high';
dc.drawImage(canvas,0,0,RW,RH,0,0,dispW,dispH);
const pw=document.createElement('div');pw.className='page-canvas-wrap';
pw.style.width='100%';
pw.appendChild(dispCanvas);wrap.appendChild(pw);

const btn=document.createElement('button');
btn.className='btn btn-outline btn-sm';
btn.style.marginRight='4px';
btn.textContent='Seite '+(page+1)+' PNG';
const p=page;
btn.onclick=()=>downloadPagePNG(p);
pngBtns.appendChild(btn);

}

document.getElementById('outputCard').style.display='block';
document.getElementById('pageCountLabel').textContent=totalPages+' Seite'+(totalPages>1?'n':'');
missing.size
?(document.getElementById('missingCharsLabel').textContent='⚠ Fehlend: '+[...missing].filter(c=>c.trim()).slice(0,15).join(' '))
:(document.getElementById('missingCharsLabel').textContent='✓ Alle Zeichen vorhanden');
document.getElementById('outputCard').scrollIntoView({behavior:'smooth',block:'nearest'});
toast('Fertig! '+totalPages+' Seite'+(totalPages>1?'n':'')+' generiert.');
}

if(cfg.mc){
ctx.save();ctx.strokeStyle=cfg.mc;ctx.lineWidth=cfg.mw||1.2;ctx.setLineDash([]);
ctx.beginPath();ctx.moveTo(A4_ML-8,4);ctx.lineTo(A4_ML-8,A4_H-4);ctx.stroke();ctx.restore();
}
if(cfg.topLine){
ctx.save();ctx.strokeStyle='rgba(60,120,200,.45)';ctx.lineWidth=1.4;ctx.setLineDash([]);
ctx.beginPath();ctx.moveTo(0,firstBaseline-lineH*2.2);ctx.lineTo(A4_W,firstBaseline-lineH*2.2);ctx.stroke();
ctx.restore();
}
for(let base=firstBaseline;base<A4_H-A4_MB+lineH*.4;base+=lineH){
ctx.save();ctx.strokeStyle=cfg.lc;ctx.lineWidth=cfg.lw||.9;ctx.setLineDash([]);
ctx.beginPath();ctx.moveTo(A4_ML-8,base);ctx.lineTo(A4_W-A4_MR,base);ctx.stroke();ctx.restore();
}
}
if(cfg.gc){
const gs=Math.round(lineH*.5);
ctx.save();ctx.strokeStyle=cfg.gc;ctx.lineWidth=cfg.gw||.7;ctx.setLineDash([]);
for(let y=0;y<=A4_H;y+=gs){ctx.beginPath();ctx.moveTo(0,y);ctx.lineTo(A4_W,y);ctx.stroke();}
for(let x=0;x<=A4_W;x+=gs){ctx.beginPath();ctx.moveTo(x,0);ctx.lineTo(x,A4_H);ctx.stroke();}
ctx.restore();
}
if(cfg.dc){
const gs=Math.round(lineH*.5);const dr=cfg.dr||1.2;
ctx.save();ctx.fillStyle=cfg.dc;
for(let y=gs;y<A4_H;y+=gs)for(let x=gs;x<A4_W;x+=gs){
ctx.beginPath();ctx.arc(x,y,dr,0,Math.PI*2);ctx.fill();
}
ctx.restore();
}
}

function applyPresetValues(p){
const set=(id,val)=>{const el=document.getElementById(id);if(el&&val!=null)el.value=val;};
set('charSize',p.charSize||22);
set('charSpacing',p.charSpacing!=null?p.charSpacing:-1);
set('wordSpacing',p.wordSpacing||0.55);
set('lineSpacingMult',p.lineSpacingMult||2.4);
set('randomMode',p.randomMode||'random');
set('paperStyle',p.paperStyle||'gn-lined');
set('inkColor',p.inkColor||'#0a0502');
const errorToggleEl=document.getElementById('errorToggle');
if(errorToggleEl){if(p.errorOn)errorToggleEl.classList.add('on');else errorToggleEl.classList.remove('on');}
const errorCtrlEl=document.getElementById('errorControls');
if(errorCtrlEl)errorCtrlEl.style.display=(p.errorOn?'block':'none');
set('errorRate',p.errorRate||1.5);
set('jitterAmt',p.jitterAmt||2.5);
set('slantDeg',p.slantDeg!=null?p.slantDeg:-5);
set('slantVar',p.slantVar||2);
const applyToggle=(swId,ctrlId,on)=>{
const sw=document.getElementById(swId);const ctrl=document.getElementById(ctrlId);
if(!sw)return;
if(on)sw.classList.add('on');else sw.classList.remove('on');
if(ctrl)ctrl.style.display=on?'block':'none';
};
applyToggle('jitterToggle','jitterControls',!!p.jitterOn);
applyToggle('slantToggle','slantControls',!!p.slantOn);
if(p.theme&&p.theme!==currentTheme)toggleTheme();
}
